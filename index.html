const Find_Prod = (array, N) => 
        {
          let product=1;

Q.10 Min or Max 

 function arrayMin(arr) 
        {
        let min=arr[0]; 
        for(let i=0; i < arr.length; i++)
        {
        if(arr[i+1] < min)
        {
         min=arr[i+1];
         }
        }
        return min;
        }

Q.11 Birthday Game 

function Birthday_Game(arr,D ,M) {
        let count=0; let N=arr.length;
        for(let i=0; i < N; i++)
        {
          let j=0; let k=i; let sum=0;
          while(j < M)
          {
            sum = sum + arr[k];
            j++; k++;
          }
          if(sum===D)
          {
            count++;
          }
        }
        return count;
  }
